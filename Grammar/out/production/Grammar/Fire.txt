Tankname Fire;

Num dist = 50;

Setup {

}
Repeat {
    Gun.turn(5);
}

When scannedRobot {
    if (Event.distance() < 50 AND Tank.energy() > 50) {
    Gun.shoot(3);
    }
    else {
        Gun.shoot(1);
    }
    Radar.scan();
}

When hitByBullet {
    Tank.turn(Utils.normalRelativeAngleDegrees(90 - (Tank.heading() - Event.heading())));

    		Tank.forward(dist);
    		dist = dist * (-1);
    		Radar.scan();
}

When hitRobot {
    Num turnGunAmt = Utils.normalRelativeAngleDegrees(Event.bearing() + Tank.heading() - Gun.heading());
    Gun.turn(turnGunAmt);
    Gun.shoot(3);
}
